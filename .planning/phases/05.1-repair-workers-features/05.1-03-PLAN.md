Rewrite `apps/api/src/workers/challenge-submission.worker.ts` to be the XP orchestrator. This worker receives `ChallengeSubmissionPayload` (userId, challengeSlug, challengeId, difficulty) and:

1. Calculates `isFirstChallenge` by counting `userXpTransaction` rows where `action = "challenge_completed"` and `userId = payload.userId`. If count === 0, it's the first.
2. Calculates `currentStreak` via `calculateStreak(userId)`.
3. Calculates XP via `calculateXPGain({ difficulty, isFirstChallenge, currentStreak })`.
4. Fires `trackChallengeCompletedServer(userId, challengeId, challengeSlug, difficulty, xpGain.total, isFirstChallenge)`.
5. Dispatches XP_AWARD jobs via a separate queue instance:
   - Always: base XP job `{ userId, challengeId, challengeSlug, xpAmount: xpGain.baseXP, action: "challenge_completed", description: "Completed ${difficulty} challenge" }`
   - If `isFirstChallenge`: bonus job `{ userId, challengeId, challengeSlug, xpAmount: xpGain.firstChallengeBonus, action: "first_challenge", description: "First challenge bonus" }`
   - If `xpGain.streakBonus > 0`: streak job `{ userId, challengeId, challengeSlug, xpAmount: xpGain.streakBonus, action: "daily_streak", description: "${currentStreak} day streak bonus" }`

Full implementation:
```typescript
import { type ChallengeSubmissionPayload, createQueue, QUEUE_NAMES } from "@kubeasy/jobs";
import type { XpAwardPayload } from "@kubeasy/jobs";
import { and, count, eq } from "drizzle-orm";
import { Worker } from "bullmq";
import { db } from "../db/index.js";
import { userXpTransaction } from "../db/schema/index.js";
import { trackChallengeCompletedServer } from "../lib/analytics-server.js";
import { calculateStreak, calculateXPGain } from "../services/xp/index.js";

export function createChallengeSubmissionWorker() {
  const redisUrl = new URL(process.env.REDIS_URL ?? "redis://localhost:6379");
  const connection = {
    host: redisUrl.hostname,
    port: Number(redisUrl.port || 6379),
    maxRetriesPerRequest: null as null,
  };

  // Queue instance for dispatching XP_AWARD jobs
  const xpAwardQueue = createQueue(QUEUE_NAMES.XP_AWARD, connection);

  return new Worker<ChallengeSubmissionPayload>(
    QUEUE_NAMES.CHALLENGE_SUBMISSION,
    async (job) => {
      const { userId, challengeSlug, challengeId, difficulty } = job.data;

      // 1. Check if this is the user's first completed challenge
      const [completedTransactions] = await db
        .select({ count: count() })
        .from(userXpTransaction)
        .where(
          and(
            eq(userXpTransaction.userId, userId),
            eq(userXpTransaction.action, "challenge_completed"),
          ),
        );
      const isFirstChallenge = (completedTransactions?.count ?? 0) === 0;

      // 2. Calculate streak
      const currentStreak = await calculateStreak(userId);

      // 3. Calculate XP amounts
      const xpGain = calculateXPGain({
        difficulty,
        isFirstChallenge,
        currentStreak,
      });

      // 4. Fire analytics (fire-and-forget style, errors logged internally)
      await trackChallengeCompletedServer(
        userId,
        challengeId,
        challengeSlug,
        difficulty,
        xpGain.total,
        isFirstChallenge,
      );

      // 5. Dispatch XP_AWARD jobs
      // Base XP always awarded
      await xpAwardQueue.add("xp-base", {
        userId,
        challengeId,
        challengeSlug,
        xpAmount: xpGain.baseXP,
        action: "challenge_completed",
        description: `Completed ${difficulty} challenge`,
      });

      // First challenge bonus
      if (isFirstChallenge && xpGain.firstChallengeBonus > 0) {
        await xpAwardQueue.add("xp-first-challenge", {
          userId,
          challengeId,
          challengeSlug,
          xpAmount: xpGain.firstChallengeBonus,
          action: "first_challenge",
          description: "First challenge bonus",
        });
      }

      // Streak bonus
      if (xpGain.streakBonus > 0) {
        await xpAwardQueue.add("xp-streak", {
          userId,
          challengeId,
          challengeSlug,
          xpAmount: xpGain.streakBonus,
          action: "daily_streak",
          description: `${currentStreak} day streak bonus`,
        });
      }
    },
    { connection, concurrency: 5 },
  );
}
```

**apps/api/src/workers/xp-award.worker.ts** -- Rewrite to perform atomic DB operations + SSE publish:

```typescript
import { QUEUE_NAMES, type XpAwardPayload } from "@kubeasy/jobs";
import { queryKeys } from "@kubeasy/api-schemas/query-keys";
import { sql } from "drizzle-orm";
import { Worker } from "bullmq";
import { db } from "../db/index.js";
import { userXp, userXpTransaction } from "../db/schema/index.js";
import { redis } from "../lib/redis.js";

export function createXpAwardWorker() {
  const redisUrl = new URL(process.env.REDIS_URL ?? "redis://localhost:6379");
  const connection = {
    host: redisUrl.hostname,
    port: Number(redisUrl.port || 6379),
    maxRetriesPerRequest: null as null,
  };

  return new Worker<XpAwardPayload>(
    QUEUE_NAMES.XP_AWARD,
    async (job) => {
      const { userId, challengeId, xpAmount, action, description } = job.data;

      // 1. Atomic userXp UPSERT (add xpAmount to totalXp)
      await db
        .insert(userXp)
        .values({ userId, totalXp: xpAmount })
        .onConflictDoUpdate({
          target: userXp.userId,
          set: {
            totalXp: sql`${userXp.totalXp} + ${xpAmount}`,
            updatedAt: new Date(),
          },
        });

      // 2. Insert userXpTransaction record
      await db.insert(userXpTransaction).values({
        userId,
        action,
        xpAmount,
        challengeId,
        description,
      });

      // 3. Publish SSE cache-invalidation event for user's XP query
      const channel = `invalidate-cache:${userId}`;
      const payload = JSON.stringify({ queryKey: queryKeys.user.xp() });
      await redis.publish(channel, payload).catch((err) => {
        console.error("[xp-award] SSE publish failed", { channel, error: String(err) });
      });
    },
    { connection, concurrency: 5 },
  );
}
```

**apps/api/src/workers/user-lifecycle.worker.ts** -- Rewrite as user-signin BullMQ Flow with FlowProducer:

The user-signin worker uses a FlowProducer to create a parent job with 3 child jobs:
- Child 1: PostHog identify (`identifyUserServer`)
- Child 2: Resend contact creation (`createResendContact`) -- returns `contactId`
- Child 3: PostHog user_signup event (`trackUserSignupServer`)

The parent job reads child 2's result and updates `user.resendContactId` in DB.

```typescript
import { QUEUE_NAMES, type UserSigninPayload } from "@kubeasy/jobs";
import { eq } from "drizzle-orm";
import { FlowProducer, Worker } from "bullmq";
import { db } from "../db/index.js";
import { user } from "../db/schema/auth.js";
import { identifyUserServer, trackUserSignupServer } from "../lib/analytics-server.js";
import { createResendContact } from "../lib/resend.js";

// Child queue names (internal, not exported)
const CHILD_QUEUES = {
  IDENTIFY: "user-signin:identify",
  RESEND_CONTACT: "user-signin:resend-contact",
  TRACK_SIGNUP: "user-signin:track-signup",
} as const;

export function createUserSigninWorker() {
  const redisUrl = new URL(process.env.REDIS_URL ?? "redis://localhost:6379");
  const connection = {
    host: redisUrl.hostname,
    port: Number(redisUrl.port || 6379),
    maxRetriesPerRequest: null as null,
  };

  const flowProducer = new FlowProducer({ connection });

  // Child worker: PostHog identify
  const identifyWorker = new Worker(
    CHILD_QUEUES.IDENTIFY,
    async (job) => {
      const { userId, email } = job.data as { userId: string; email: string };
      await identifyUserServer(userId, { email });
    },
    { connection, concurrency: 5 },
  );

  // Child worker: Resend contact creation
  const resendContactWorker = new Worker(
    CHILD_QUEUES.RESEND_CONTACT,
    async (job) => {
      const { email, userId } = job.data as { email: string; userId: string };
      const result = await createResendContact({ email, userId });
      return result; // { contactId: string } -- available to parent via childrenValues
    },
    { connection, concurrency: 5 },
  );

  // Child worker: PostHog track signup
  const trackSignupWorker = new Worker(
    CHILD_QUEUES.TRACK_SIGNUP,
    async (job) => {
      const { userId, provider, email } = job.data as { userId: string; provider: string; email: string };
      await trackUserSignupServer(userId, provider as "github" | "google" | "microsoft", email);
    },
    { connection, concurrency: 5 },
  );

  // Parent worker: captures resend contactId and stores it
  const parentWorker = new Worker<UserSigninPayload>(
    QUEUE_NAMES.USER_SIGNIN,
    async (job) => {
      const { userId } = job.data;

      // childrenValues is a Record<queueName:jobId, returnValue>
      const childrenValues = await job.getChildrenValues();

      // Find the resend-contact child result
      const resendResult = Object.entries(childrenValues).find(
        ([key]) => key.startsWith(CHILD_QUEUES.RESEND_CONTACT),
      );

      if (resendResult) {
        const { contactId } = resendResult[1] as { contactId: string };
        if (contactId) {
          await db
            .update(user)
            .set({ resendContactId: contactId })
            .where(eq(user.id, userId));
        }
      }
    },
    { connection, concurrency: 5 },
  );

  // Return all workers for batch close on SIGTERM
  return {
    workers: [parentWorker, identifyWorker, resendContactWorker, trackSignupWorker],
    async addFlow(data: UserSigninPayload) {
      const { userId, email, provider } = data;
      await flowProducer.add({
        name: "user-signin-flow",
        queueName: QUEUE_NAMES.USER_SIGNIN,
        data,
        children: [
          {
            name: "identify",
            queueName: CHILD_QUEUES.IDENTIFY,
            data: { userId, email },
          },
          {
            name: "resend-contact",
            queueName: CHILD_QUEUES.RESEND_CONTACT,
            data: { email, userId },
          },
          {
            name: "track-signup",
            queueName: CHILD_QUEUES.TRACK_SIGNUP,
            data: { userId, provider, email },
          },
        ],
      });
    },
    async close() {
      await Promise.all([
        parentWorker.close(),
        identifyWorker.close(),
        resendContactWorker.close(),
        trackSignupWorker.close(),
        flowProducer.close(),
      ]);
    },
  };
}
```

**REVISED approach for user-lifecycle.worker.ts** -- Use a standard Worker that internally runs all 3 operations using `better-all`:

```typescript
import { QUEUE_NAMES, type UserSigninPayload } from "@kubeasy/jobs";
import { eq } from "drizzle-orm";
import { Worker } from "bullmq";
import { all } from "better-all";
import { db } from "../db/index.js";
import { user } from "../db/schema/auth.js";
import { identifyUserServer, trackUserSignupServer } from "../lib/analytics-server.js";
import { createResendContact } from "../lib/resend.js";

export function createUserSigninWorker() {
  const redisUrl = new URL(process.env.REDIS_URL ?? "redis://localhost:6379");
  const connection = {
    host: redisUrl.hostname,
    port: Number(redisUrl.port || 6379),
    maxRetriesPerRequest: null as null,
  };

  return new Worker<UserSigninPayload>(
    QUEUE_NAMES.USER_SIGNIN,
    async (job) => {
      const { userId, email, provider } = job.data;

      // Run all 3 operations in parallel with better-all
      const { resendResult } = await all({
        async identify() {
          await identifyUserServer(userId, { email });
        },
        async resendResult() {
          try {
            return await createResendContact({ email, userId });
          } catch (err) {
            // Log and continue -- Resend failure should not block other operations
            console.error("[user-signin] Resend contact creation failed", {
              userId,
              error: String(err),
            });
            return null;
          }
        },
        async trackSignup() {
          await trackUserSignupServer(
            userId,
            provider as "github" | "google" | "microsoft",
            email,
          );
        },
      });

      // Store Resend contactId on user record if available
      if (resendResult?.contactId) {
        await db
          .update(user)
          .set({ resendContactId: resendResult.contactId })
          .where(eq(user.id, userId));
      }
    },
    { connection, concurrency: 5 },
  );
}
```

Use this REVISED approach (standard Worker + better-all) instead of the FlowProducer approach. It's simpler, works with `queue.add()` in auth.ts, and achieves the same parallelism.
