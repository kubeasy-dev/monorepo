4. Create `packages/logger/src/index.ts`:
   ```typescript
   import pino from "pino";

   const isDev = process.env.NODE_ENV !== "production";

   /** Only scalar values — matches the existing logger API contract */
   export type LogAttributes = Record<string, string | number | boolean>;

   export const logger = pino({
     level: process.env.LOG_LEVEL ?? (isDev ? "debug" : "info"),
     transport: isDev
       ? { target: "pino-pretty", options: { colorize: true } }
       : undefined,
   });
   ```

   Note on API compatibility: pino's native API is `logger.info(obj, msg)` or `logger.info(msg)`. The existing logger uses `logger.info(message, attributes)`. Pino supports `logger.info(attributes, message)` natively. Callers that currently do `logger.info("msg", { key: "val" })` will need to swap argument order to `logger.info({ key: "val" }, "msg")` — OR create wrapper functions. Since the research says "pino accepts objects as second argument natively" but pino actually uses `logger.info(mergingObject, message)` format, implement wrapper functions to preserve the existing `(message, attributes?)` signature:

   ```typescript
   import pino from "pino";

   const isDev = process.env.NODE_ENV !== "production";

   export type LogAttributes = Record<string, string | number | boolean>;

   const pinoInstance = pino({
     level: process.env.LOG_LEVEL ?? (isDev ? "debug" : "info"),
     transport: isDev
       ? { target: "pino-pretty", options: { colorize: true } }
       : undefined,
   });

   export const logger = {
     debug: (message: string, attributes?: LogAttributes) =>
       attributes ? pinoInstance.debug(attributes, message) : pinoInstance.debug(message),
     info: (message: string, attributes?: LogAttributes) =>
       attributes ? pinoInstance.info(attributes, message) : pinoInstance.info(message),
     warn: (message: string, attributes?: LogAttributes) =>
       attributes ? pinoInstance.warn(attributes, message) : pinoInstance.warn(message),
     error: (message: string, attributes?: LogAttributes) =>
       attributes ? pinoInstance.error(attributes, message) : pinoInstance.error(message),
   };
   ```
